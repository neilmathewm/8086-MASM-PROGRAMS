.MODEL SMALL
.STACK 100H 
PRINT MACRO Q
LEA DX,Q
MOV AH,09H
INT 21H
ENDM 
.DATA
MSG1 DB 10,13,"ENTER THE NO :  $"  
MSG2 DB 10,13,"SQUARE ROOT  :  $" 
MSG3 DB 10,13,"NOT PERFECT SQUARE!!$" 
.CODE

MOV AX,@DATA
MOV DS,AX

PRINT MSG1

MOV AH,01H     ;READ HIGHER BIT
INT 21H  
MOV BH,AL 

MOV AH,01H          ;READ LOWER BIT
INT 21H 
MOV BL,AL

SUB BH,30H
SUB BL,30H
MOV CX,0000H        ;COUNT=0
  
OUTER:
    MOV AX,CX
    MUL AX          ;SQUARE AX*AX
    AAM             ;ascii adjust mult
    CMP AX,BX
    JZ PRINT_ROOT      
    INC CX          ;INCREEMENT COUNT
    ;CMP CX,BX 
     CMP CL,10      ;TILL 9^2 IS REQUIRED 10^2 IS 3 BYTE
JNZ OUTER
PRINT MSG3          ;IF NO ROOT
JZ QUIT

PRINT_ROOT: 

PRINT MSG2
MOV AH,02H
MOV DL,CL
ADD DL,30H
INT 21H


QUIT:
.EXIT
END
